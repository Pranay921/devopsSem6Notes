Image creation in detail ->

- mkdir nginx-app
- cd nginx-app

- notepad index.html
----->
<!DOCTYPE html>
<html>
<head>
    <title>Medium Level Docker App</title>
</head>
<body>
    <h1>Welcome to My Custom Docker Nginx App</h1>
    <p>This is a medium-level Docker project.</p>
</body>
</html>

- notepad default.conf
----->
server {
    listen 80;
    server_name localhost;

    location / {
        root /usr/share/nginx/html;
        index index.html;
    }
}

- notepad Dockerfile
----->
FROM nginx:alpine

COPY index.html /usr/share/nginx/html/index.html
COPY default.conf /etc/nginx/conf.d/default.conf

EXPOSE 80

ren Dockerfile.txt Dockerfile

--> Dir
--> docker build -t custom-nginx:v1 .
--> docker run -d -p 8080:80 --name nginx-container custom-nginx:v1

http://localhost:8080

mkdir node-app
cd node-app

apt update
apt install nano -y

Create app.js
----->
const express = require("express");
const app = express();

app.get("/", (req, res) => {
  res.send("Docker Node App Running!");
});

app.listen(3000, "0.0.0.0", () => {
  console.log("Server running on port 3000");
});

Create package.json
----->
{
  "name": "node-docker-app",
  "version": "1.0.0",
  "main": "app.js",
  "dependencies": {
    "express": "^4.18.2"
  }
}

Create Dockerfile
----->
FROM node:18-alpine
WORKDIR /app
COPY package.json .
RUN npm install
COPY app.js .
EXPOSE 3000
CMD ["node", "app.js"]

VERIFY FILES ->
Dir
app.js
package.json
